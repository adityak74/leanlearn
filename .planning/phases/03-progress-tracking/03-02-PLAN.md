<task type="auto">
  <name>Task 2: Progress calculation logic</name>
  <files>app/server.ts</files>
  <action>
    Inside the completion route:
    1. Find the `courseId` associated with the activity.
    2. Count total `required` activities for that course.
    3. Count completed `required` activities for this user in this course.
    4. Calculate percentage: `Math.floor((completed / total) * 100)`.
    5. Update or insert into `courseProgress` table.
    6. If percentage is 100, set `completedAt` timestamp.
  </action>
  <verify>
    <automated>Verify database record in course_progress after completion call.</automated>
  </verify>
  <done>Course progress is updated accurately in the database.</done>
</task>

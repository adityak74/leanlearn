<task type="auto">
  <name>Task 1: Setup Drizzle Schema and D1 Initial Migration</name>
  <files>app/db/schema.ts, drizzle.config.ts, package.json</files>
  <action>
    Define the Better Auth schema using Drizzle ORM and generate/apply the initial D1 migration.
    - Install `drizzle-orm` and `drizzle-kit`.
    - Create `app/db/schema.ts` with Better Auth tables: `user`, `session`, `account`, `verification`.
    - Configure `drizzle.config.ts` for Cloudflare D1.
    - Run `npx drizzle-kit generate` to create the migration.
    - Run `npx wrangler d1 migrations apply DB --local` to apply the migration locally.
  </action>
  <verify>
    <automated>npx wrangler d1 execute DB --local --command "SELECT name FROM sqlite_master WHERE type='table';" | grep -E "user|session|account|verification"</automated>
  </verify>
  <done>D1 database contains the necessary tables for Better Auth.</done>
</task>

<task type="auto">
  <name>Task 2: Implement getAuth Factory and Mount Routes</name>
  <files>app/server.ts</files>
  <action>
    Implement the `getAuth(c)` factory and mount Better Auth routes in the Hono application.
    - Define `getAuth(c)` factory using `betterAuth`.
    - Use `drizzleAdapter` with the D1 binding and the defined schema.
    - Configure Google OAuth provider with environment variables.
    - Implement `advanced.runInBackground` to use `c.executionCtx.waitUntil(fn())`.

  </action>

  </verify>
  <done>Better Auth is integrated into Hono and ready to handle auth requests.</done>
